class FewShotTemplate:

        self,

        version_id: str,
        test_inputs: List[str],

    ) -> Dict:
        """æµ‹è¯•æç¤ºè¯ç‰ˆæœ¬"""

        if not prompt:
            raise ValueError(f"Prompt {prompt_id} not found")

            (v for v in prompt.versions if v.version_id == version_id),

        """é»˜è®¤è¯„åˆ†å‡½æ•°ï¼ˆç®€åŒ–ç‰ˆï¼‰"""
        # åŸºäºé•¿åº¦å’ŒåŸºæœ¬æŒ‡æ ‡
